For version: Ubuntu 18.04 LTS

A. Install Git Client

  1. sudo apt-get install git

B. Install Go (version 1.15.x)

Installation Instructions
  1. $ wget https://go.dev/dl/go1.15.linux-amd64.tar.gz
  2. $ sudo tar -C /usr/local/ -xzf go1.15.linux-amd64.tar.gz
  3. $vim ~/.profile
  4. export PATH=$PATH:/usr/local/go/bin
  5. :wq
  6. $ source ~/.profile
  7. $ go
  8. $ go version

C. Install Docker version 18.03 or later

Installation Instructions
  1. sudo apt-get update
  2. sudo apt-get install \
    ca-certificates \
    curl \
    gnupg \
    lsb-release
  3. y
  4. y
  5. curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg
  6. echo \ "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu \ $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
  7. sudo apt-get update
  8. sudo apt-get install docker-ce docker-ce-cli containerd.io
  9. y
  10. sudo docker run hello-world

D. Install SoftHSM

  1. git clone https://github.com/opendnssec/SoftHSMv2.git
  2. sudo apt-get install openssl
  3. sudo apt-get install libssl-dev
  4. ls
  5. cd SoftHSMv2
  6. sudo apt install automake
  7. sudo apt install autoconf
  8. sudo apt install libtool
  9. *sudo apt install libtool-ltdl-devel (RHEL/CentOS)
  10. sudo apt install pkg-config
  11. sh autogen.sh
  12. ./configure
  13. make
  14. sudo make install
  15. cd ..

E. Install jq

  1. sudo apt-get install jq
  2. y
  3. jq--version

F. Install cURL

  1. sudo apt -y install curl

G. Clone the Hyperledger Fabric Source

Installation Instructions
  1. git clone https://github.com/hyperledger/fabric
  2. softhsm2-util --init-token --slot 0 --label "ForFabric" --so-pin 1234 --pin 98765432
  3. ls
  4. cd fabric
  5. make gotools
  6. cd ..
